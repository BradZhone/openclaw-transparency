Quick Start

# Install
pip install openclaw-transparency

# Track AI agent actions
from openclaw_transparency_mvp import TransparencyLayer

t = TransparencyLayer("MyAgent", tier="free")

# Track actions
t.track_action("file_read", "config.py", "Config loaded")
t.track_action("api_call", "/api/data", "Data fetched")

# Create checkpoint
t.create_checkpoint("Init complete", ["config.py"], [])

# Get summary
summary = t.end_session()
print(summary)
